const express = require('express');
const app = express();
const path = require('path');

// Define the absolute path to your project folder
const projectDir = 'C:\\Users\\HP\\OneDrive\\mini project sem5'; // Use double backslashes or path.join

// Serve static files like CSS, JS, images from this folder
app.use(express.static(projectDir));

// Routes
app.get('/home', (req, res) => {
  res.sendFile(path.join(projectDir, 'home.html'));
});
app.get('/user', (req, res) => {
  if (req.session.user) {
    res.json(req.session.user);
  } else {
    res.status(401).json({ message: 'Not logged in' });
  }
});
app.get('/signup', (req, res) => {
  res.sendFile(path.join(projectDir, 'signup.html'));
});

app.get('/login', (req, res) => {
  res.sendFile(path.join(projectDir, 'login.html'));
});
app.post('/logout', (req, res) => {
  req.session.destroy((err) => {
    if (err) {
      return res.status(500).json({ message: 'Logout failed' });
    }
    res.clearCookie('connect.sid'); // if using express-session
    res.json({ message: 'Logged out successfully' });
  });
});

app.get('/changepass', (req, res) => {
  res.sendFile(path.join(projectDir, 'changepass.html'));
});
